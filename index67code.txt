// api/index.js
import { WebSocketDurable } from './durable/websocket.js';
import { handleDataManagement } from './data/data-manager.js';
import fs from 'fs';
import path from 'path';

// Dynamically read data.json from api/data/data.json
let data = {};
try {
  const dataPath = path.join(process.cwd(), 'api', 'data', 'data.json');
  data = JSON.parse(fs.readFileSync(dataPath, 'utf-8'));
} catch (err) {
  console.error('Failed to read data.json', err);
}

// Dynamically read main-page.html from root
let mainPage = '';
try {
  const mainPagePath = path.join(process.cwd(), 'main-page.html');
  mainPage = fs.readFileSync(mainPagePath, 'utf-8');
} catch (err) {
  console.error('Failed to read main-page.html', err);
}

const liquidWebSocket = new WebSocketDurable();

export default async function handler(req, res) {
  const url = new URL(req.url, `http://${req.headers.host}`);

  // Serve main-page.html for root and /liquid
  if (url.pathname === '/' || url.pathname === '/liquid') {
    res.setHeader('Content-Type', 'text/html; charset=utf-8');
    return res.status(200).send(mainPage);
  }

  // Dashboard endpoint
  if (url.pathname === '/dashboard' || url.pathname === '/dash') {
    const socketMapsJson = {
      trackerCount: 0,
      liquids: liquidWebSocket.getUsers()
    };
    return res.status(200).json({
      trackers: socketMapsJson.trackerCount,
      liquids: socketMapsJson.liquids.map(u => u.username)
    });
  }

  // Data endpoints: /data, /json, /serverdata
 if (
  url.pathname === '/data' ||
  url.pathname === '/json' ||
  url.pathname === '/serverdata'
) {
  return res.status(200).json(data); // <-- returns your real data.json
}

  // Manage endpoint
  if (url.pathname === '/manage') {
    return handleDataManagement(req, res);
  }

  // WebSocket endpoint
  if (url.pathname === '/websocket') {
    return liquidWebSocket.fetch(req, res);
  }

  // Friendly 404 for everything else
  return res.status(404).json({
    status: 404,
    error: 'NotFound',
    message:
      "The URL you're looking for does not exist. Some common URLs here at liquid-theta.vercel.app are '/data' and '/dashboard'. Were you perhaps looking for those?"
  });
}
